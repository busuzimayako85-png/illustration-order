function calculate() { const form = document.getElementById('orderForm'); let rawSubtotal = 0; // 割引前の小計 let activeCardCount = 0; // ベースが選択されているイラストの枚数 let allBasesSelected = true; const cards = document.querySelectorAll('.illustration-card'); if (cards.length === 0) { allBasesSelected = false; } cards.forEach(card => { const id = card.id.split('_')[1]; const baseRadio = document.querySelector(`input[name="base_${id}"]:checked`); if (!baseRadio) { allBasesSelected = false; return; } activeCardCount++; const bgRadio = document.querySelector(`input[name="bg_${id}"]:checked`).value; rawSubtotal += PRICES.base[baseRadio.value]; rawSubtotal += PRICES.bg[bgRadio]; ['faceDiff', 'poseDiff', 'clothDiff'].forEach(type => { const checkbox = document.getElementById(`${type}Check_${id}`); const qtyInput = document.getElementById(`${type}Qty_${id}`); if (checkbox && checkbox.checked && qtyInput) { rawSubtotal += PRICES.diff[type] * parseInt(qtyInput.value); } }); }); // ─── 複数枚割引の判定ロジック ─── let discountRate = 0; if (activeCardCount === 2) { discountRate = 0.05; // 2枚で 5% OFF } else if (activeCardCount >= 3) { discountRate = 0.10; // 3枚以上で 10% OFF } const discountAmount = Math.floor(rawSubtotal * discountRate); // 割引額 const finalTotal = rawSubtotal - discountAmount; // 最終合計金額 // 表示のコントロール document.getElementById('totalDisplay').innerText = finalTotal.toLocaleString() + " 円"; // 送信ボタンの制御 const submitBtn = document.getElementById('submitBtn'); if (isStopMode) { submitBtn.disabled = true; submitBtn.innerText = "ORDERS CURRENTLY CLOSED"; } else { submitBtn.disabled = (activeCardCount === 0); submitBtn.innerText = submitBtn.disabled ? "SELECT BASE TYPE" : "PROCEED TO HEARING"; } }
